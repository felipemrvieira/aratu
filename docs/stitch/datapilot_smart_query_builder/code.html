SELECT
  u.id,
  u.email,
  u.created_at,
  COUNT(o.id) AS paid_orders
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE 
  u.created_at >= DATE_TRUNC('month', CURRENT_DATE)
  AND o.status = 'paid'
GROUP BY 
  u.id, u.email, u.created_at
ORDER BY 
  paid_orders DESC;
